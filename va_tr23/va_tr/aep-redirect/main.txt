#include <sgx_urts.h>
#include "Enclave/encl_u.h"
#include <sys/mman.h>
#include <signal.h>
#include "libsgxstep/enclave.h"
#include "libsgxstep/debug.h"
#include "libsgxstep/pt.h"

#define DBG_ENCL        1
int retval1 = 0;
int retval2 = 0;
int main( int argc, char **argv )
{
    

    sgx_enclave_id_t enclave_id1,enclave_id2;
    sgx_launch_token_t token={0};
    int updated=0;
    sgx_create_enclave( "./Enclave/encl.so", /*debug=*/DBG_ENCL,&token, &updated, &enclave_id1, NULL );
    sgx_create_enclave( "./Enclave/encl.so", /*debug=*/DBG_ENCL,&token, &updated, &enclave_id2, NULL );
    enclave_dummy_call(enclave_id1,&retval1);
    enclave_dummy_call(enclave_id2,&retval2);
    printf("r1=%d,r2=%d",retval1,retval2);
    getchar();
    sgx_destroy_enclave(enclave_id1);
    sgx_destroy_enclave(enclave_id2);
    return 0;
}
